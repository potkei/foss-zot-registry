#!/usr/bin/env bash
# =============================================================================
# SECRET SCAN — Gitleaks: detects secrets in code and git history
# This file is stored as .sh.txt — translate to .sh via onboard skill.
# =============================================================================
set -euo pipefail

ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/../../.." && pwd)"
source "${ROOT_DIR}/.cicd/scan-versions.env"

mkdir -p "${ROOT_DIR}/reports"

echo "=== SECRET SCAN (Gitleaks ${GITLEAKS_VERSION}) ==="

docker run --rm \
  -v "${ROOT_DIR}":/src \
  "zricethezav/gitleaks:${GITLEAKS_VERSION}" \
  detect \
    --source=/src \
    --report-format=json \
    --report-path=/src/reports/gitleaks.json \
    --verbose

FINDINGS=$(jq 'length' "${ROOT_DIR}/reports/gitleaks.json" 2>/dev/null || echo 0)
if [ "${FINDINGS}" -gt 0 ]; then
  echo "SECRET SCAN FAILED — ${FINDINGS} secret(s) detected!"
  jq -r '.[] | "  \(.RuleID): \(.File):\(.StartLine)"' "${ROOT_DIR}/reports/gitleaks.json"
  exit 1
fi

echo "=== SECRET SCAN PASSED — no secrets found ==="
