#!/usr/bin/env bash
# =============================================================================
# INIT — First-run setup after creating repo from template
# Translates .sh.txt scripts to executable .sh, creates required directories,
# and validates prerequisites are installed.
# Run once: ./init.sh
# =============================================================================
set -euo pipefail

ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"

echo "==================================================="
echo " FOSS Boilerplate — First-Run Setup"
echo "==================================================="
echo ""

# -------------------------------------------------------------------------
# 1. Validate prerequisites
# -------------------------------------------------------------------------
echo "--- Checking prerequisites ---"

MISSING=()

# Accept docker or podman as the container runtime
DOCKER_CMD=""
if command -v docker >/dev/null 2>&1; then
  DOCKER_CMD="docker"
elif command -v podman >/dev/null 2>&1; then
  DOCKER_CMD="podman"
  echo "  INFO: Using podman as container runtime."
else
  MISSING+=("docker or podman")
fi

command -v git >/dev/null 2>&1 || MISSING+=("git")

if [ ${#MISSING[@]} -gt 0 ]; then
  echo "ERROR: Missing required tools: ${MISSING[*]}"
  echo "Install them and re-run ./init.sh"
  exit 1
fi

# jq — required for JSON parsing in build/scan/release scripts.
# Minimum version: 1.6. If missing or too old, build from .tools/jsonq using Go.
_jq_version_ok() {
  command -v jq >/dev/null 2>&1 || return 1
  local ver major minor
  ver=$(jq --version 2>/dev/null | sed 's/jq-//')
  major=$(echo "$ver" | cut -d. -f1)
  minor=$(echo "$ver" | cut -d. -f2)
  [ "${major:-0}" -gt 1 ] || { [ "${major:-0}" -eq 1 ] && [ "${minor:-0}" -ge 6 ]; }
}

if _jq_version_ok; then
  echo "  jq $(jq --version) — OK."
else
  # Determine output binary name (jq.exe on Windows/Git Bash, jq elsewhere)
  JQ_BIN="jq"
  case "$(uname -s)" in MINGW*|MSYS*|CYGWIN*) JQ_BIN="jq.exe" ;; esac
  JQ_OUT="${ROOT_DIR}/.tools/bin/${JQ_BIN}"

  if command -v jq >/dev/null 2>&1; then
    echo "  WARN: jq $(jq --version) is too old (need >= 1.6) — building replacement from .tools/jsonq."
  else
    echo "  INFO: jq not found — building from .tools/jsonq using Go (GOPROXY=${GOPROXY:-direct})."
  fi

  if command -v go >/dev/null 2>&1; then
    if (cd "${ROOT_DIR}/.tools/jsonq" && go build -o "${JQ_OUT}" . 2>&1); then
      echo "  INFO: Built .tools/bin/${JQ_BIN} successfully."
      echo "  INFO: Add to your PATH: export PATH=\"\$PATH:${ROOT_DIR}/.tools/bin\""
      export PATH="${ROOT_DIR}/.tools/bin:${PATH}"
    else
      echo "  WARN: Go build failed. Install jq >= 1.6 manually or check your GOPROXY setting."
    fi
  else
    echo "  WARN: Go not installed. Cannot auto-build jq."
    echo "        Install Go (https://go.dev/dl/) then re-run ./init.sh — it will build jq automatically."
    echo "        Or install jq >= 1.6 directly: https://jqlang.github.io/jq/download/"
  fi
fi

# Optional tools — warn but don't block
command -v cosign  >/dev/null 2>&1 || echo "  WARN: cosign not found (needed for image signing)"
command -v helm    >/dev/null 2>&1 || echo "  WARN: helm not found (needed for Helm chart operations)"

echo "  All required tools present."
echo ""

# -------------------------------------------------------------------------
# 2. Create required directories
# -------------------------------------------------------------------------
echo "--- Creating directories ---"

for dir in patches reports docs helm; do
  if [ ! -d "${ROOT_DIR}/${dir}" ]; then
    mkdir -p "${ROOT_DIR}/${dir}"
    echo "  Created: ${dir}/"
  else
    echo "  Exists:  ${dir}/"
  fi
done

# Ensure reports is gitignored but dir exists
touch "${ROOT_DIR}/reports/.gitkeep"
touch "${ROOT_DIR}/patches/.gitkeep"

echo ""

# -------------------------------------------------------------------------
# 3. Translate .txt script files to executable originals
# -------------------------------------------------------------------------
echo "--- Translating script files (.sh.txt -> .sh) ---"

TRANSLATED=0

find "${ROOT_DIR}" -name "*.sh.txt" | sort | while read -r f; do
  target="${f%.txt}"
  if [ ! -f "$target" ]; then
    cp "$f" "$target"
    chmod +x "$target"
    echo "  Translated: ${target#${ROOT_DIR}/}"
  else
    echo "  Exists:     ${target#${ROOT_DIR}/}"
  fi
done

find "${ROOT_DIR}" -name "*.py.txt" | sort | while read -r f; do
  target="${f%.txt}"
  if [ ! -f "$target" ]; then
    cp "$f" "$target"
    chmod +x "$target"
    echo "  Translated: ${target#${ROOT_DIR}/}"
  fi
done

echo ""

# -------------------------------------------------------------------------
# 4. Make root scripts executable
# -------------------------------------------------------------------------
echo "--- Setting executable permissions ---"

for script in build.sh init.sh; do
  if [ -f "${ROOT_DIR}/${script}" ]; then
    chmod +x "${ROOT_DIR}/${script}"
    echo "  chmod +x: ${script}"
  fi
done

echo ""

# -------------------------------------------------------------------------
# 5. Verify Docker is running
# -------------------------------------------------------------------------
echo "--- Checking container runtime daemon ---"
if ${DOCKER_CMD} info >/dev/null 2>&1; then
  echo "  ${DOCKER_CMD} daemon is running."
else
  echo "  WARN: ${DOCKER_CMD} daemon not reachable. Start it before building."
fi

echo ""
echo "==================================================="
echo " Init complete!"
echo ""
echo " Next steps:"
echo "   1. Run the onboard skill to configure for your FOSS project"
echo "   2. Or manually update Dockerfiles and package.json"
echo "   3. Run: ./build.sh          (source build)"
echo "   4. Run: ./build.sh --scan   (security scans)"
echo "==================================================="
